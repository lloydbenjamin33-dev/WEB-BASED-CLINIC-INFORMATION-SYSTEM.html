<script>
  let currentRole = 'patient';
  let signupRole  = 'patient';

  function setRole(r, el) {
    currentRole = r;
    document.querySelectorAll('#loginPage .role-tab').forEach(t => t.classList.remove('active'));
    el.classList.add('active');
  }
  function setSignupRole(r, el) {
    signupRole = r;
    document.querySelectorAll('#signupPage .role-tab').forEach(t => t.classList.remove('active'));
    el.classList.add('active');
    document.getElementById('signupSpecField').style.display = r === 'doctor' ? 'block' : 'none';
  }

  function showPage(id) {
    document.querySelectorAll('.auth-wrap').forEach(p => p.classList.remove('active'));
    if (id) document.getElementById(id).classList.add('active');
  }

  function hideAllAuth() {
    document.querySelectorAll('.auth-wrap').forEach(p => p.classList.remove('active'));
  }

  function doLogin() {
    const email = document.getElementById('loginEmail').value.trim();
    const pass  = document.getElementById('loginPass').value.trim();
    if (!email || !pass) { notify('Please enter email and password.'); return; }
    hideAllAuth();
    document.querySelectorAll('.layout').forEach(l => l.classList.remove('active'));
    if (currentRole === 'patient') { document.getElementById('patientLayout').classList.add('active'); }
    else if (currentRole === 'doctor') { document.getElementById('doctorLayout').classList.add('active'); }
    else { document.getElementById('adminLayout').classList.add('active'); }
    notify('Welcome back! Signed in successfully.');
  }

  function doSignup() {
    notify('Account created! You can now sign in.');
    showPage('loginPage');
  }

  function doLogout() {
    document.querySelectorAll('.layout').forEach(l => l.classList.remove('active'));
    document.querySelectorAll('.auth-wrap').forEach(p => p.classList.remove('active'));
    document.getElementById('loginPage').classList.add('active');
    notify('Signed out successfully.');
  }

  // page sections
  function showSection(prefix, sectionId, navEl) {
    // pages
    document.querySelectorAll(`#${prefix === 'p' ? 'patientLayout' : prefix === 'd' ? 'doctorLayout' : 'adminLayout'} .page-content`).forEach(p => p.classList.remove('active'));
    document.getElementById(sectionId).classList.add('active');
    // nav items
    if (navEl) {
      const layout = prefix === 'p' ? 'patientLayout' : prefix === 'd' ? 'doctorLayout' : 'adminLayout';
      document.querySelectorAll(`#${layout} .nav-item`).forEach(n => n.classList.remove('active'));
      navEl.classList.add('active');
    }
    // topbar title
    const titles = {
      pDash:'Dashboard',pAppts:'My Appointments',pBook:'Book Appointment',pMessages:'Messages',pProfile:'My Profile',
      dDash:'Dashboard',dAppts:'Appointments',dSchedule:'My Schedule',dMessages:'Messages',
      aDash:'Dashboard',aAppts:'Appointments',aPatients:'Patients',aDoctors:'Doctors',aMessages:'Messages',aReports:'Reports',aUsers:'User Management'
    };
    const titleEl = document.getElementById(prefix+'PageTitle');
    if (titleEl && titles[sectionId]) titleEl.textContent = titles[sectionId];
  }

  function notify(msg) {
    const old = document.querySelector('.notif');
    if (old) old.remove();
    const el = document.createElement('div');
    el.className = 'notif';
    el.textContent = msg;
    document.body.appendChild(el);
    setTimeout(() => el.remove(), 3000);
  }

  function openModal(id) { document.getElementById(id).classList.add('open'); }
  function closeModal(id) { document.getElementById(id).classList.remove('open'); }

  // close modal on bg click
  document.querySelectorAll('.modal-bg').forEach(bg => {
    bg.addEventListener('click', e => { if (e.target === bg) bg.classList.remove('open'); });
  });

  function bookAppointment() {
    const date = document.getElementById('bookDate').value;
    const slot = document.getElementById('bookSlot').value;
    if (!date) { notify('Please select a date.'); return; }
    notify(`Appointment booked for ${date} at ${slot}!`);
  }

  function toggleDay(el) {
    el.classList.toggle('taken');
  }

  function toggleModalSpec() {
    const r = document.getElementById('modalRole').value;
    document.getElementById('modalSpecField').style.display = r === 'Doctor' ? 'block' : 'none';
  }

  // set today as default booking date
  document.addEventListener('DOMContentLoaded', () => {
    const d = document.getElementById('bookDate');
    if (d) d.valueAsDate = new Date();
  });
</script>
