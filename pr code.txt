}

function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 2800);
}

function bookAppointment() {
  const patient = document.getElementById('appt-patient').value;
  const doctor = document.getElementById('appt-doctor').value;
  const date = document.getElementById('appt-date').value;
  const time = document.getElementById('appt-time').value;
  const type = document.getElementById('appt-type').value;
  if (!patient || !doctor || !date) { showToast('Please fill all required fields'); return; }
  const newId = 'APT-0' + (appointments.length + 23);
  const parts = doctor.split(' — ');
  appointments.unshift({ id:newId, patient:patient.split(' (')[0], doctor:parts[0], date:date, time:time, type:type, status:'Pending' });
  renderAppointments();
  closeModal('appointmentModal');
  showToast('Appointment booked — confirmation sent to patient & doctor');
}

function registerPatient() {
  const fn = document.getElementById('pt-fname').value;
  const ln = document.getElementById('pt-lname').value;
  const ph = document.getElementById('pt-phone').value;
  if (!fn || !ln) { showToast('Please enter patient name'); return; }
  const newId = 'PT-0' + (249 + patients.length);
  patients.unshift({ id:newId, name:`${fn} ${ln}`, phone:ph||'—', history:'None', appts:0, status:'New' });
  renderPatients();
  closeModal('patientModal');
  showToast(`Patient ${fn} ${ln} registered successfully`);
}

function addDoctor() {
  const fn = document.getElementById('dr-fname').value;
  const ln = document.getElementById('dr-lname').value;
  const spec = document.getElementById('dr-spec').value;
  if (!fn || !ln) { showToast('Please enter doctor name'); return; }
  const selected = [...document.querySelectorAll('#day-selector .slot.selected')].map(s=>s.textContent);
  const color = COLORS[doctors.length % COLORS.length];
  doctors.push({ id:'DR-00'+(doctors.length+1), name:`Dr. ${fn} ${ln}`, spec, appts:0, color, slots:['09:00','10:30','14:00'], days:selected.join(', ')||'TBD', initials:`${fn[0]}${ln[0]}` });
  renderDoctors();
  closeModal('doctorModal');
  showToast(`Dr. ${fn} ${ln} added to the system`);
}

function cancelAppt(id, btn) {
  const a = appointments.find(x => x.id===id);
  if (a) { a.status = 'Cancelled'; renderAppointments(); showToast(`Appointment ${id} cancelled`); }
}

  box.value = '';
  showToast('Reply sent successfully');
}
